const { resolve } = require('path');
 
const REPO = resolve('.');
 
const { execFile } = require('child_process');
 
class Git {
  executeGit(cmd, args) {
    return new Promise((res, reject) => {
      execFile(cmd, args, { cwd: REPO }, (err, stdout) => {
        if (err) {
          reject(err);
        }
        console.log(typeof stdout);
        res(stdout.toString());
      });
    });
  }
 
  gitHistory(page = 1, size = 10) {
    const offset = (page - 1) * size;
 
    return this.executeGit('git', [
      'log',
      '--pretty=format:%H%x09%an%x09%ad%x09%s',
      '--date=iso',
      '--skip',
      offset,
      '-n',
      size,
    ])
      .then(data => this.splitData(data, this.parseHistoryItem));
  }
 
  splitData(data, method) {
    return data.split('\n')
      .filter(Boolean)
      .map(method);
  }
 
  parseHistoryItem(line) {
    const [hash, author, timestamp, msg] = line.split('\t');
 
    return {
      hash,
      author,
      timestamp,
      msg,
    };
  }
 
  parseFileTreeItem(line) {
    const [info, path] = line.split('\t');
    const [, type, hash] = info.split(' ');
 
    return {
      type,
      hash,
      path,
    };
  }
 
  gitFileTree(hash, path) {
    const params = ['ls-tree', hash];
    path && params.push(path);
    return this.executeGit('git', params)
      .then(data => this.splitData(data, this.parseFileTreeItem));
  }
 
  gitFileContent(hash) {
    return this.executeGit('git', ['show', hash]);
  }
}
 
const git = new Git();
 
module.exports = {
  Git,
  git,
};
